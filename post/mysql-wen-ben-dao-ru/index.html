MySQL如何导入文本文件

load data infile 'path\test.txt' into table table_name;

  1. 查看系统文件路径：
show global variables like "%datadir%";  #C:\ProgramData\MySQL\MySQL Server 5.7\Data\
show variables like "%secure%";  #secure_file_priv
  1. 修改C:\ProgramData\MySQL\MySQL Server 5.7\Data\my.ini文件中的secure-file-priv=''。
  2. 需要先创建好一个Table
  3. 插入的文本中不需要有列名
use cp;
create table score(
sid int(4),
student_id int(4),
corse_id int(4),
number int(10)
);
load data infile 'D:\Learn\MySQL\inputTest.txt' into table score;
